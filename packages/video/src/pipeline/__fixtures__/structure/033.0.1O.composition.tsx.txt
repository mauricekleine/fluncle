// Track: Eventide — Monrroe (Liquid Tones, 2014)
// Vehicle: tidal cell field — voronoi3 cellular web; amber walls burn in warm-dark
//   at the crossing hour, blazing at the 7s bass peak
// Texture family: fluent | Grain: grainChemicalDye | Register: abstract
// Concept: The full frame is a living cellular membrane at eventide — voronoi3
//   cells evolve forward on a monotonic z-phase (never returning), amber walls
//   widen on each bass hit, blazing amber-cream at the 7s bass-peak climax.
//   Depth: upper-center landing (0.5, 0.68 UV) where cells are densest and
//   the last light gathers; corners fall into warm dark.

import React from "react";
import {
  CloseCard,
  GLSL,
  ShaderLayer,
  TrackAudio,
  TypePlate,
  paletteMix,
  useJourney,
  type NostalgicCosmosProps,
} from "../cosmos";

// oxlint-disable-next-line sort-keys
const frag = `
${GLSL.hash}
${GLSL.valueNoise}
${GLSL.fbm}
${GLSL.paletteRamp}
${GLSL.noise3}
${GLSL.filmGrain}
${GLSL.grainFamilies}
${GLSL.tonemap}

void main() {
  vec2 uv = gl_FragCoord.xy / u_res;

  // Global translation: constant one-way audio-free clock
  vec2 drift = vec2(0.055, 0.095) * u_time;
  // Monotonic z-phase: cells roil forward, never return
  float zPhase = u_time * 0.19;

  // Voronoi3 cellular field
  float cellScale = 3.3;
  vec3 vor = voronoi3(vec3((uv + drift) * cellScale, zPhase));
  // vor.x = sqrt(F1), vor.y = sqrt(F2), vor.z = cell hash
  float edge = vor.y - vor.x;  // 0 on wall, increases into cell interior
  float cellHash = vor.z;

  // Wall width: in-place structural binding (bass + kick + drop)
  float wallW = 0.036 + u_bass * 0.026 + u_kickHit * 0.014 + u_audioDrop * 0.046;
  float wall = 1.0 - smoothstep(0.0, wallW, edge);

  // Landing point: upper-center of 9:16 frame (uv y=0 bottom, y=1 top)
  vec2 lp = vec2(0.5, 0.68);
  float distLP = length((uv - lp) * vec2(0.80, 1.20));
  float landing = exp(-distLP * 1.85);

  // Cell interior: warm-dark (paletteRamp near 0), slightly lifted at landing
  float cellVariance = (cellHash - 0.5) * 0.05;
  float interiorT = max(0.0, cellVariance + landing * 0.11 + u_mid * 0.05);
  vec3 interiorCol = paletteRamp(interiorT);

  // Wall color: amber (0.3) through cream (1.0) via paletteRamp
  float wallLum = wall * (0.40 + u_audioSwell * 0.28 + u_audioDrop * 0.44);
  wallLum *= (0.50 + landing * 0.50);
  vec3 wallCol = paletteRamp(clamp(wallLum * 1.65, 0.0, 1.0));

  // Combine: amber walls over warm-dark interiors; extra hot cluster at landing
  vec3 col = mix(interiorCol, wallCol, wall);
  col += wallCol * wall * 0.48 * landing;

  // Exposure envelope: energy drives overall brightness
  float exposure = 0.76 + u_energy * 0.42 + landing * 0.20;
  col *= exposure;

  // Grain: chemical dye — color-film on organic material
  // Texture binding: amount rises with treble (hat tails / sparkle)
  GrainOpts grain = grainChemicalDye();
  grain.amount = 0.035 + u_treble * 0.011;
  col = filmGrain(col, uv, u_time, grain);

  // Filmic soft-clip and 8-bit dither
  col = acesFilmic(col);
  col = dither8(col, uv);

  gl_FragColor = vec4(col, 1.0);
}
`;

const TidalCellField: React.FC<NostalgicCosmosProps> = ({ audio, palette, seed, track }) => {
  const { phase, phaseProgress } = useJourney();
  const mixedPalette = paletteMix(palette.swatches);

  return (
    <>
      <ShaderLayer
        fragmentShader={frag}
        palette={mixedPalette}
        seed={seed}
        beatGrid={audio.beatGrid}
        energyCurve={audio.energyCurve}
        bassCurve={audio.bassCurve}
        midCurve={audio.midCurve}
        trebleCurve={audio.trebleCurve}
        fluxCurve={audio.fluxCurve}
        subCurve={audio.subCurve}
        kickCurve={audio.kickCurve}
        snareCurve={audio.snareCurve}
        airCurve={audio.airCurve}
        downbeats={audio.downbeats}
        onsets={audio.onsets}
        dropMs={audio.dropMs}
        reactivity={{
          drop: { fallMs: 5500, holdMs: 1200, peakTimeMs: 7000, riseMs: 2800 },
        }}
        bloom={{ intensity: 0.62, radius: 0.85, threshold: 0.68 }}
      />
      <TypePlate
        dimInk={mixedPalette.glow}
        identityInSec={1.3}
        ink={mixedPalette.ink}
        telemetryInSec={2.0}
        track={track}
      />
      <TrackAudio audio={audio} />
      {phase === "arrive" && (
        <CloseCard
          palette={{ accent: mixedPalette.accent, ink: mixedPalette.ink }}
          progress={phaseProgress}
        />
      )}
    </>
  );
};

export default TidalCellField;
